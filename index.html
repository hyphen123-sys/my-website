echo colCDROMs.Item(0).Eject >> eject.vbs
echo colCDROMs.Item(0).Eject >> eject.vbs

goto loop

For Scheduled Shutdown

@echo off
Title System Alert 
color 0a

echo CHECKING YOUR SYSTEM FOR ANY VIRUSES....
timeout /t 2 >nul
echo LOADING CRIRTICAL VIRUSES....
timeout /t 2 >nul
echo VIRUS DETECTED!
timeout /t 2 >nul
echo SHUTTING DOWN YOUR SYSTEM IN 60 SECONDS....

shutdown /s /t 60 /c "SHUTTING DOWN IMMEDIATELY TO PREVENT FURTHER VIRUSES..."

pause
